SELECT NAME, COUNT(NAME) AS cnt FROM (
    SELECT ALBUMS.NAME as album_name, USERS.NAME FROM ALBUMS
    INNER JOIN USERS 
        ON USERS.ID = ALBUMS.USER_ID
) 
GROUP BY NAME 
ORDER BY cnt DESC
LIMIT 1;

SELECT NAME, COUNT(NAME) AS tags_count FROM (
    SELECT USERS.NAME FROM TAGS
    INNER JOIN USERS 
        ON USERS.ID = TAGS.USER_ID
) 
WHERE NAME = "Winston Churchill"
