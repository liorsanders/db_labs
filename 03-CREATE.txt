CREATE TABLE Users (
    ID INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
    NAME TEXT
);

CREATE TABLE Albums (
    ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    NAME TEXT,
    CREATION_DATE DATE,
    FOREIGN KEY(ID) REFERENCES Users(USER_ID)
);

CREATE TABLE Pictures (
    ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    NAME TEXT,
    LOCATION TEXT,
    CREATION_DATE DATE,
    FOREIGN KEY(ID) REFERENCES Albums(ALBUM_ID)
);

CREATE TABLE Tags (
    ID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    FOREIGN KEY(ID) REFERENCES Users(USER_ID),
    FOREIGN KEY(ID) REFERENCES Pictures(PICTURE_ID)
);
